! This text tileset lists the tiles that either aren't generated by
! basecchar, or that need overriding even in ASCII.

! Furthest backgrounds need to be opaque.
unexplored area: bgblack black regular ' '
the floor of a room: bgblack gray regular '.'
corridor: bgblack gray regular '#'

! Dark room, lit corridor.
sub unlit the floor of a room: bgblack darkgray regular '.'
sub lit corridor: bgblack white regular '#'

! Everyone hates the default boulder tile.
boulder: gray '0'

! Stairs and portals have red backgrounds (that show through).
! TODO: Express these with a generator, to insulate against changes in core.
staircase up: bgred white '<'
staircase down: bgred white '>'
ladder up: bgred yellow '<'
ladder down: bgred yellow '>'
long staircase up: bgred yellow underlined '<'
long staircase down: bgred yellow underlined '>'
magic portal: bgred bright_magenta '^'
vibrating square: bgred yellow '^'

! Rogue level graphical overrides.
sub rogue open doors ?: gray '+'
sub rogue doorway: gray '+'
sub rogue *staircase *: gray '%'
sub rogue gold piece: gray '*'
! As in NitroHack, override the most common food
! TODO: Override the others too?
sub rogue corpse; sub rogue food ration: gray ':'

! The invisible monster symbol isn't defined in core.
invisible monster: bright_blue 'I'

! Nor are the brandings.
remembered as stepped on: disturb invisible
remembered as locked: red invisible
remembered as unlocked: green invisible
remembered as trapped: bgcyan samefg invisible

monster is tame: bgblue samefg invisible
monster is peaceful: bgbrown samefg invisible
monster is detected: bgmagenta samefg invisible
monster is ridden: samefg invisible
